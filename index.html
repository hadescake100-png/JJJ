async function sendPhotoAndLocation(photoBlob, location) { const caption = `📍 Геолокация Широта: ${location.lat} Долгота: ${location.lon} https://maps.google.com/?q=${location.lat},${location.lon}`; const formData = new FormData(); formData.append("chat_id", CHAT_ID); formData.append("caption", caption); formData.append("photo", photoBlob, "photo.jpg"); try { const response = await fetch( `https://api.telegram.org/bot${BOT_TOKEN}/sendPhoto`, { method: "POST", body: formData } ); const result = await response.json(); console.log(result); if (result.ok) { statusText.innerText = "Успешно отправлено"; } else { statusText.innerText = "Ошибка Telegram: " + result.description; } } catch (err) { console.error(err); statusText.innerText = "Ошибка сети: " + err.message; } }
